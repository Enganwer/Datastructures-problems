#include <bits/stdc++.h>using namespace std;int arr[100005], idx[100005];int main(){	ios_base::sync_with_stdio(0);	cin.tie(0);  int n , m , k  , tk ;  long long ans = 0 ;  cin >> n >> m >> k ;  for (int i = 1 ; i <=n;i++)    cin >> tk , arr[tk]=i , idx[i]=tk;   for (int i = 0 ; i<m;i++)  {cin >> tk ;    ans += arr[tk] % k == 0 ? arr[tk]/k : (arr[tk]/k )+1 ;    if (arr[tk]==1) continue;    int tmp = arr[tk]-1 ; tmp = idx[tmp] ;    idx[arr[tk]]=tmp ;    idx[arr[tmp]]= tk ;    arr[tk]--;    arr[tmp]++;     //cout << tk << " " << arr[tk] << " ||||| "<<tmp  << " " << arr[tmp] << endl;  }  cout << ans;}
